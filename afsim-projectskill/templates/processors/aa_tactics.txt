# AA Tactics Processor (from demos/shooter/)
# State-machine based air-to-air engagement logic.
# States: DETECTED → ENGAGEABLE → ENGAGE → (fire)

include_once weapons/aam/aa_launch_computer.txt

processor AA_TACTICS WSF_TASK_PROCESSOR
   script_variables
      string SENSOR_NAME               = "";
      string SENSOR_TRACK_MODE         = "TRACK";
      string WEAPON_NAME               = "";
      double REQUIRED_TRACK_QUALITY    = 0.49;
      int SALVO_SIZE                   = 1;
   end_script_variables

   # Check if weapons are still active on this track
   script bool FiringNow()
      return (WeaponsActiveFor(TRACK.TrackId()) > 0);
   end_script

   # Check if target is in weapon intercept envelope
   script bool InInterceptEnvelopeOf(WsfWeapon aWeapon)
      return Local_AA_LaunchComputer(TRACK, aWeapon, 0.0);
   end_script

   # Count available weapons
   script int WeaponsAvailable()
      WsfWeapon weapon = PLATFORM.Weapon(WEAPON_NAME);
      if (weapon.IsTurnedOn())
         return weapon.QuantityRemaining();
      return 0;
   end_script

   # Fire weapon at track
   script bool LaunchWeapon()
      WsfWeapon weapon = PLATFORM.Weapon(WEAPON_NAME);
      bool canIntercept = (weapon.IsTurnedOn() && WeaponsAvailable() >= SALVO_SIZE
                           && InInterceptEnvelopeOf(weapon));
      if (!canIntercept) return false;
      weapon.CueToTarget(TRACK);
      return FireAt(TRACK, "SHOOT", weapon, SALVO_SIZE);
   end_script

   # Begin tracking target with sensor
   script bool Engage()
      if (WeaponsAvailable() < 1) return false;
      WsfSensor sensor = PLATFORM.Sensor(SENSOR_NAME);
      if (!sensor.IsValid()) return false;
      if (sensor.ActiveRequestCount(SENSOR_TRACK_MODE) >=
          sensor.MaximumRequestCount(SENSOR_TRACK_MODE))
         return false;
      return StartTracking(TRACK, "TRACK", sensor, SENSOR_TRACK_MODE);
   end_script

   # ── State Machine ────────────────────────────────────
   evaluation_interval DETECTED 2.0 sec
   state DETECTED
      next_state ENGAGEABLE
         return (TRACK.BelievedAlive() && !TRACK.IFF_Friend() &&
                 (PLATFORM == PLATFORM.Commander() ||
                  TasksReceivedFor(TRACK.TrackId()) > 0));
      end_next_state
   end_state

   evaluation_interval ENGAGEABLE 2.0 sec
   state ENGAGEABLE
      next_state ENGAGE
         Engage();
         return (TRACK.BelievedAlive() &&
                 InInterceptEnvelopeOf(PLATFORM.Weapon(WEAPON_NAME)) &&
                 TRACK.TrackQuality() > REQUIRED_TRACK_QUALITY);
      end_next_state
   end_state

   evaluation_interval ENGAGE 2.0 sec
   state ENGAGE
      next_state DETECTED
         if (!TRACK.BelievedAlive() || !TrackingNow()) return true;
         bool launched = false;
         if (TRACK.BelievedAlive() && !FiringNow() &&
             InInterceptEnvelopeOf(PLATFORM.Weapon(WEAPON_NAME)))
            launched = LaunchWeapon();
         return launched;
      end_next_state
   end_state
end_processor
