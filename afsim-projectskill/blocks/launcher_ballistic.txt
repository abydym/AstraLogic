# Generic ground-based ballistic missile launcher (SRBM / MRBM).
# Uses LAUNCH_WEAPON_PROCESSOR for script-driven launch timing.
# Each launcher holds a single SSM weapon.
#
# REQUIRES (via file_path ../base_types):
#   weapons/ssm/red_srbm_1.txt  (or variant)
#   processors/launch_weapon_processor.txt
# =============================================================================

# --- Short Range: RED_SRBM_1_LAUNCHER ---
platform_type RED_SRBM_1_LAUNCHER WSF_PLATFORM
   icon Scud_Launcher
   side red
   mover WSF_GROUND_MOVER end_mover
   weapon red_srbm_1 RED_SRBM_1
      quantity 1
   end_weapon
   processor launch_weapon_processor LAUNCH_WEAPON_PROCESSOR
      script_variables
         WEAPON_NAME = "red_srbm_1";
      end_script_variables
   end_processor
end_platform_type
